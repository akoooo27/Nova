# Projects Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add ChatGPT-style **Projects** — a user-owned container that groups conversations and injects project-level instructions into every turn of its chats.

**Architecture:** New `Project` aggregate in `Chat.Domain.Projects`, a nullable `ProjectId` grouping reference on `ChatThread`, project instructions folded into the turn system prompt at the existing `ContextBuilder`/`PersonalizationSystemPrompt` seam, CQRS commands/queries in `Chat.Application`, EF write-side + Dapper read-side in `Chat.Infrastructure`, and FastEndpoints in `Chat.Api`. Reads go through Dapper readers; writes through the EF repository. Deleting a project cascades to its chats via a set-based repository delete.

**Tech Stack:** .NET, EF Core (Npgsql, snake_case), Dapper, Mediator (source generator), FastEndpoints, FluentValidation, ErrorOr, xUnit.

## Global Constraints

- Use `Mediator` (`ICommand`/`IQuery`/`ICommandHandler`/`IQueryHandler`) — never MediatR.
- HTTP via FastEndpoints; mutations use `BaseEndpoint<TRequest,TResponse>` + `SendErrorOrAsync`.
- Command/query results are `ErrorOr<T>`; validators are FluentValidation `AbstractValidator<T>`.
- Value objects are `sealed record` with a private ctor, `public const int MaxLength`, `static ErrorOr<T> Create(string?)`, `static T FromDatabase(...)` (throws `DomainException`), and `override string ToString()`.
- Domain stays persistence-ignorant; no cross-aggregate navigation properties and **no** DB-level FK between `chats` and `projects`.
- Aggregate roots call `builder.Ignore(x => x.DomainEvents)` and `.ValueGeneratedNever()` on the id in their EF config.
- Ids are `Guid.CreateVersion7()`.
- Commits use the repo convention `feat(chat): …` / `docs(chat): …` with **no** `Co-Authored-By` trailer. Work stays on the current `replicate-chatgpt-projects` branch.
- Do not touch the MassTransit version. Do not modify unrelated tests.
- Turn-pipeline rules hold: Agent-Framework types stay in `Chat.Infrastructure/Agents`; `ChatTurnOrchestrator` stays sequencing-only; `IContextBuilder.BuildAsync` keeps its signature (new deps arrive by constructor); `TurnRequested` carries ids only.

    {
        if (value == Guid.Empty)

        return new ProjectId(value);
    }

    public override string ToString() => Value.ToString();
}

using ErrorOr;

namespace Chat.Domain.Projects.ValueObjects;

public sealed record ProjectName
{
    public const int MaxLength = 120;

    public string Value { get; }

    private ProjectName(string value) => Value = value;

    public static ErrorOr<ProjectName> Create(string? value)
    {
        string trimmed = value?.Trim() ?? string.Empty;

        if (string.IsNullOrWhiteSpace(trimmed))
        {
            return Error.Validation("ProjectName.Required", "Project name is required.");
        }

        if (trimmed.Length > MaxLength)
        {
            return Error.Validation("ProjectName.TooLong", $"Project name cannot exceed {MaxLength} characters.");
        }

        return new ProjectName(trimmed);
    }

    public static ProjectName FromDatabase(string value)
    {
        if (string.IsNullOrWhiteSpace(value) || value.Length > MaxLength)
            throw new DomainException("Database contained an invalid project name.");

        return new ProjectName(value);
    }

    public override string ToString() => Value;
}

namespace Chat.Domain.Projects.ValueObjects;

public sealed partial record ProjectTheme
{
    public string Value { get; }

    private ProjectTheme(string value) => Value = value;

    public static ErrorOr<ProjectTheme> Create(string? value)
    {
        string trimmed = value?.Trim() ?? string.Empty;

        if (string.IsNullOrWhiteSpace(trimmed))

        }

        if (!HexColor().IsMatch(trimmed))
        {
            return Error.Validation("ProjectTheme.Invalid", "Project theme must be a hex color like #F6C543.");

        return new ProjectTheme($"#{trimmed.TrimStart('#').ToUpperInvariant()}");
    }

    public static ProjectTheme FromDatabase(string value)
    {
        if (string.IsNullOrWhiteSpace(value) || !HexColor().IsMatch(value))
            throw new DomainException("Database contained an invalid project theme.");

        return new ProjectTheme(value);
    }

    private static partial Regex HexColor();

        ProjectName name,
        ProjectInstructions? instructions,
        ProjectEmoji? emoji,

using Chat.Application.Projects.Errors;
using Chat.Domain.Chats;
using Chat.Domain.Projects;
using Chat.Domain.Projects.ValueObjects;
using Chat.Domain.Shared;

using ErrorOr;

using Mediator;

using Shared.Application.Authentication;

namespace Chat.Application.Projects.Commands.DeleteProject;

internal sealed class DeleteProjectHandler(
    IUserContext userContext,
    IProjectRepository projects,
    IChatRepository chats,
    IUnitOfWork unitOfWork) : ICommandHandler<DeleteProjectCommand, ErrorOr<Success>>
{
    public async ValueTask<ErrorOr<Success>> Handle(DeleteProjectCommand command, CancellationToken cancellationToken)
    {
        ErrorOr<UserId> userIdResult = UserId.Create(userContext.UserId);
        ErrorOr<ProjectId> projectIdResult = ProjectId.Create(command.ProjectId);

        List<Error> errors = [];

        if (userIdResult.IsError)
        {
            errors.AddRange(userIdResult.Errors);
        }

        if (projectIdResult.IsError)
        {
            errors.AddRange(projectIdResult.Errors);
        }

        if (errors.Count > 0)
        {
            return errors;
        }

        Project? project = await projects.GetByIdAsync(projectIdResult.Value, userIdResult.Value, cancellationToken);

        if (project is null)
        {
            return ProjectOperationErrors.ProjectNotFound(projectIdResult.Value);
        }

        // Cascade: delete the project's chats (set-based) first, then remove the project.

        await chats.DeleteByProjectAsync(projectIdResult.Value, userIdResult.Value, cancellationToken);

        await unitOfWork.SaveChangesAsync(cancellationToken);

        return Result.Success;
    }

using Chat.Application.Abstractions.Database;
using Chat.Application.Chats.Errors;
using Chat.Application.Chats.Results;
using Chat.Application.Projects.Errors;
using Chat.Domain.Chats;
using Chat.Domain.Chats.ValueObjects;

using Chat.Domain.Shared;

using ErrorOr;

using Mediator;

using Shared.Application.Authentication;

namespace Chat.Application.Chats.Commands.MoveChatToProject;

internal sealed class MoveChatToProjectHandler(
    IUserContext userContext,
    IChatRepository chats,
    IProjectRepository projects,
    IUnitOfWork unitOfWork,
    IDateTimeProvider dateTimeProvider) : ICommandHandler<MoveChatToProjectCommand, ErrorOr<ChatThreadResult>>
{
    public async ValueTask<ErrorOr<ChatThreadResult>> Handle(MoveChatToProjectCommand command, CancellationToken cancellationToken)
    {
        ErrorOr<UserId> userIdResult = UserId.Create(userContext.UserId);
        ErrorOr<ChatId> chatIdResult = ChatId.Create(command.ChatId);

        List<Error> errors = [];

        if (userIdResult.IsError)
        {
            errors.AddRange(userIdResult.Errors);
        }

        if (chatIdResult.IsError)
        {
            errors.AddRange(chatIdResult.Errors);
        }

        ProjectId? targetProjectId = null;

        if (command.ProjectId is { } rawProjectId)
        {
            ErrorOr<ProjectId> projectIdResult = ProjectId.Create(rawProjectId);

            if (projectIdResult.IsError)
            {
                errors.AddRange(projectIdResult.Errors);
            }
            else
            {
                targetProjectId = projectIdResult.Value;
            }
        }

        if (errors.Count > 0)
        {
            return errors;
        }

        ChatThread? thread = await chats.GetByIdAsync(chatIdResult.Value, userIdResult.Value, cancellationToken);

        if (thread is null)
        {
            return ChatOperationErrors.ChatNotFound(chatIdResult.Value);
        }

        if (targetProjectId is not null)
        {
            Project? project = await projects.GetByIdAsync(targetProjectId, userIdResult.Value, cancellationToken);

            if (project is null)
            {
                return ProjectOperationErrors.ProjectNotFound(targetProjectId);
            }
        }

        ErrorOr<Success> moveResult = thread.MoveToProject(targetProjectId, dateTimeProvider.UtcNow);

        if (moveResult.IsError)
        {
            return moveResult.Errors;
        }

        await unitOfWork.SaveChangesAsync(cancellationToken);

        return thread.ToResult();
    }
}

using Chat.Application.Chats.Queries.GetChats;
using Chat.Application.Projects.Queries;
using Chat.Domain.Projects.ValueObjects;
using Chat.Domain.Shared;

using Dapper;

using Npgsql;

namespace Chat.Infrastructure.Projects.Readers;

internal sealed class ProjectReader(NpgsqlDataSource dataSource) : IProjectReader
{
    private const string ListSql = """
                                   select
                                       id           as "Id",
                                       name         as "Name",
                                       instructions as "Instructions",
                                       emoji        as "Emoji",
                                       theme        as "Theme",
                                       created_at   as "CreatedAt",
                                       updated_at   as "UpdatedAt"
                                   from projects
                                   where user_id = @UserId
                                   order by updated_at desc, id desc;
                                   """;

    private const string DetailSql = """
                                     select
                                         id           as "Id",
                                         name         as "Name",
                                         instructions as "Instructions",
                                         emoji        as "Emoji",
                                         theme        as "Theme",
                                         created_at   as "CreatedAt",
                                         updated_at   as "UpdatedAt"
                                     from projects
                                     where id = @ProjectId and user_id = @UserId;

                                     select
                                         id           as "Id",
                                         title        as "Title",
                                         pinned_at    as "PinnedAt",
                                         is_archived  as "IsArchived",
                                         is_temporary as "IsTemporary",
                                         created_at   as "CreatedAt",
                                         updated_at   as "UpdatedAt"
                                     from chats
                                     where user_id = @UserId
                                       and project_id = @ProjectId
                                       and is_temporary = false
                                     order by (pinned_at is null), pinned_at desc, updated_at desc, id desc;
                                     """;

    public async Task<IReadOnlyList<ProjectSummaryReadModel>> ListAsync(UserId userId, CancellationToken cancellationToken)
    {
        await using NpgsqlConnection connection = await dataSource.OpenConnectionAsync(cancellationToken);

        CommandDefinition command = new(ListSql, new { UserId = userId.Value }, cancellationToken: cancellationToken);

        IEnumerable<ProjectRow> rows = await connection.QueryAsync<ProjectRow>(command);

        return rows.Select(ToSummary).ToArray();
    }

    public async Task<ProjectDetailReadModel?> GetAsync(ProjectId projectId, UserId userId, CancellationToken cancellationToken)
    {
        await using NpgsqlConnection connection = await dataSource.OpenConnectionAsync(cancellationToken);

        CommandDefinition command = new
        (
            DetailSql,
            new { ProjectId = projectId.Value, UserId = userId.Value },
            cancellationToken: cancellationToken
        );

        await using SqlMapper.GridReader grid = await connection.QueryMultipleAsync(command);

        ProjectRow? project = await grid.ReadSingleOrDefaultAsync<ProjectRow>();

        if (project is null)
        {
            return null;
        }

        ChatRow[] chatRows = (await grid.ReadAsync<ChatRow>()).ToArray();

        ChatSummaryReadModel[] chats = chatRows
            .Select(row => new ChatSummaryReadModel
            (
                Id: row.Id,
                Title: row.Title,
                IsPinned: row.PinnedAt is not null,
                PinnedAt: row.PinnedAt,
                IsArchived: row.IsArchived,
                IsTemporary: row.IsTemporary,
                CreatedAt: row.CreatedAt,
                UpdatedAt: row.UpdatedAt
            ))
            .ToArray();

        return new ProjectDetailReadModel(ToSummary(project), chats);
    }

    private static ProjectSummaryReadModel ToSummary(ProjectRow row) => new
    (
        Id: row.Id,
        Name: row.Name,
        Instructions: row.Instructions,
        Emoji: row.Emoji,
        Theme: row.Theme,
        CreatedAt: row.CreatedAt,
        UpdatedAt: row.UpdatedAt
    );

    private sealed record ProjectRow
    (
        Guid Id,
        string Name,
        string? Instructions,
        string? Emoji,
        string? Theme,
        DateTime CreatedAt,
        DateTime UpdatedAt
    );

    private sealed record ChatRow
    (
        Guid Id,
        string Title,
        DateTime? PinnedAt,
        bool IsArchived,
        bool IsTemporary,
        DateTime CreatedAt,
        DateTime UpdatedAt
    );
}

- [ ] **Step 3: Create the `CreateProject` endpoint**

```csharp
using Chat.Api.Endpoints;
using Chat.Api.Endpoints.Projects.Responses;
using Chat.Application.Projects.Commands.CreateProject;
using Chat.Application.Projects.Results;

using ErrorOr;

using FastEndpoints;

using Mediator;

using Shared.Api.Infrastructure;

namespace Chat.Api.Endpoints.Projects.CreateProject;

internal sealed record Request(string Name, string? Instructions, string? Emoji, string? Theme);

internal sealed class Endpoint(ISender sender) : Endpoint<Request>
{
    public const string RouteName = "Chat.Projects.Create";

    public override void Configure()
    {
        Post("/projects");
        Version(1);

        Options(builder => builder.WithName(RouteName));

        Description(descriptor =>
        {
            descriptor
                .WithSummary("Create Project")
                .WithDescription("Creates a project owned by the authenticated user.")
                .Produces<ProjectResponse>(StatusCodes.Status201Created)
                .ProducesProblemDetails(StatusCodes.Status400BadRequest, "application/json")
                .ProducesProblemDetails(StatusCodes.Status401Unauthorized, "application/json")
                .WithTags(CustomTags.Projects);
        });
    }

    public override async Task HandleAsync(Request request, CancellationToken ct)
    {
        CreateProjectCommand command = new(request.Name, request.Instructions, request.Emoji, request.Theme);

        ErrorOr<ProjectResult> result = await sender.Send(command, ct);

        if (result.IsError)
        {
            await Send.ResultAsync(CustomResults.Problem(result));
            return;
        }

        ProjectResponse response = ProjectResponse.From(result.Value);

        await Send.ResultAsync(TypedResults.Created($"/v1/projects/{response.Id}", response));
    }
}
```
