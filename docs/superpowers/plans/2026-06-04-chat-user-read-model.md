# Chat User Read Model Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Consume IdentityIngress user lifecycle events in Chat API and maintain a Chat.Infrastructure-owned user read model keyed by Auth0 ids.

**Architecture:** IdentityIngress publishes shared identity event contracts. Chat.Infrastructure registers MassTransit consumers for those contracts and stores projected user state in `ChatDbContext`. Authenticated request identity reads raw Auth0 `sub` so request-time identity and message-time identity use the same id.

**Tech Stack:** .NET 10, EF Core, Npgsql, MassTransit RabbitMQ with EF inbox/outbox, source-generated `LoggerMessage`, Auth0 JWT bearer authentication.

```csharp
namespace Chat.Infrastructure.Users;

internal sealed class UserReadModel
{
    private UserReadModel()
    {
    }

    private UserReadModel
    (
        string provider,
        string providerUserId,
        DateTimeOffset observedAt
    )
    {
        Id = Guid.NewGuid();
        Provider = provider;
        ProviderUserId = providerUserId;
        FirstObservedAt = observedAt;
        LastObservedAt = observedAt;
    }

    public Guid Id { get; private set; }

    public string Provider { get; private set; } = string.Empty;

    public string ProviderUserId { get; private set; } = string.Empty;

    public string? Email { get; private set; }

    public bool? EmailVerified { get; private set; }

    public string? Name { get; private set; }

    public DateTimeOffset FirstObservedAt { get; private set; }

    public DateTimeOffset LastObservedAt { get; private set; }

    public bool IsDeleted { get; private set; }

    public DateTimeOffset? DeletedAt { get; private set; }

    public static UserReadModel Create
    (
        string provider,
        string providerUserId,
        DateTimeOffset observedAt
    ) => new(provider, providerUserId, observedAt);

    public void ApplyProfile
    (
        string? email,
        bool? emailVerified,
        string? name,
        DateTimeOffset observedAt
    )
    {
        Email = email;
        EmailVerified = emailVerified;
        Name = name;
        LastObservedAt = observedAt;
        IsDeleted = false;
        DeletedAt = null;
    }

    public void MarkDeleted(DateTimeOffset deletedAt)
    {
        LastObservedAt = deletedAt;
        IsDeleted = true;
        DeletedAt = deletedAt;
    }
}
```

```csharp
using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Metadata.Builders;

namespace Chat.Infrastructure.Users.Configurations;

internal sealed class UserReadModelConfiguration : IEntityTypeConfiguration<UserReadModel>
{
    public void Configure(EntityTypeBuilder<UserReadModel> builder)
    {
        builder.ToTable("users");

        builder.HasKey(user => user.Id);

        builder.Property(user => user.Id)
            .ValueGeneratedNever();

        builder.Property(user => user.Provider)
            .HasMaxLength(UserReadModelLimits.ProviderMaxLength)
            .IsRequired();

        builder.Property(user => user.ProviderUserId)
            .HasMaxLength(UserReadModelLimits.ProviderUserIdMaxLength)
            .IsRequired();

        builder.Property(user => user.Email)
            .HasMaxLength(UserReadModelLimits.EmailMaxLength);

        builder.Property(user => user.Name)
            .HasMaxLength(UserReadModelLimits.NameMaxLength);

        builder.Property(user => user.FirstObservedAt)
            .IsRequired();

        builder.Property(user => user.LastObservedAt)
            .IsRequired();

        builder.Property(user => user.IsDeleted)
            .IsRequired();

        builder.HasIndex(user => new { user.Provider, user.ProviderUserId })
            .IsUnique();
    }
}
```

- [ ] **Step 1: Create the registration consumer**

Create `src/services/Chat/Chat.Infrastructure/Users/Consumers/UserRegisteredConsumer.cs`:

```csharp
using Chat.Infrastructure.Database;

using MassTransit;

using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Logging;

using Shared.Contracts.IdentityIngress.Events;

namespace Chat.Infrastructure.Users.Consumers;

internal sealed partial class UserRegisteredConsumer(
    ChatDbContext db,
    ILogger<UserRegisteredConsumer> logger)
    : IConsumer<UserRegistered>
{
    public async Task Consume(ConsumeContext<UserRegistered> context)
    {
        UserRegistered message = context.Message;

        UserReadModel? user = await db.Users.SingleOrDefaultAsync
        (
            candidate => candidate.Provider == message.Provider &&
                         candidate.ProviderUserId == message.ProviderUserId,
            context.CancellationToken
        );

        if (user is null)
        {
            user = UserReadModel.Create(message.Provider, message.ProviderUserId, message.OccurredAt);
            db.Users.Add(user);
        }

        user.ApplyProfile(message.Email, message.EmailVerified, message.Name, message.OccurredAt);

        await db.SaveChangesAsync(context.CancellationToken);

        LogUserRegisteredProjected(message.EventId, message.Provider, message.ProviderUserId);
    }

    [LoggerMessage(EventId = 1, Level = LogLevel.Information,
        Message = "Projected registered identity user {Provider}:{ProviderUserId} from event {EventId}")]
    private partial void LogUserRegisteredProjected(string eventId, string provider, string providerUserId);
}
```
